Main object properties
======================
xmin
	Slider with input -> number
xmax
	Slider with input -> number
xlog
	Slider with input (1 to disable) -> number
xlabelType
	Radio (KaTeX compatible LaTeX or Plaintext) -> string "plaintext", "tex", "mathtex"
xlabel
	Input -> string

ymin
	Slider with input -> number
ymax
	Slider with input -> number
ylog
	Slider with input (1 to disable) -> number
ylabelType
	Radio (KaTeX compatible LaTeX or Plaintext) -> string "plaintext", "tex", "mathtex"
ylabel
	Input -> string

backgroundColor
	Color picker -> hex color
grid
	Dropdown -> string: "none", "both", "horizontal", "vertical"
gridThickness
	Slider with input -> number
gridColor
	Color picker -> hex color
gridHalf
	Checkbox -> boolean
gridHalfColor
	Color picker -> hex color

legend
	Checkbox -> boolean
legendCoords
	Two inputs -> array [x, y]

xAxis
	Checkbox -> boolean
xAxisColor
	Color picker -> hex color
xAxisThickness
	Slider with input -> number
xAxisArrow
	Dropdown -> string: "left", "right", "none", "both"

yAxis
	Checkbox -> boolean
yAxisColor
	Color picker -> hex color
yAxisThickness
	Slider with input -> number
yAxisArrow
	Dropdown -> string: "top", "bottom", "none", "both"
